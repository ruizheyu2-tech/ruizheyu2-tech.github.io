#转速信息
TARGET_SPEED=`cat ./target_speed.txt`

TARGET_MONITOR_WIDTH=$((${TARGET_PERIOD}*3/2/8))

#屏蔽破损靶片的延迟和展宽在这里
#sheild breoken target parameters ,

#find  target 0
TARGET_NUM0_SELECT_DELAY=$((${TARGET_PERIOD}*40/72/8))
TARGET_NUM0_SELECT_WIDTH=$((${TARGET_PERIOD}*16/72/8))

#delay for next target 0 position (delay twice, 21T-20)
